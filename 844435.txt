        <script type="text/javascript">
               
               function feedback()
               {
                   var feedback = document.getElementById("fd").value;
                   var trainer = document.getElementById("tname").value;
                   var feedbackrate="";
                   var participant = document.getElementById("pname").value;
                   
                   if(feedback == "Excellent")
                   {
                       feedbackrate = 5;
                   }else if(feedback == "Very Good"){
                       feedbackrate = 4;
                   }else if(feedback == "Good"){
                       feedbackrate = 3;
                   }else if(feedback == "Average"){
                       feedbackrate = 2;
                   }else if(feedback == "Below Average"){
                       feedbackrate = 1;
                   }else{
                       feedbackrate = 0;
                   }
                   document.getElementById("result").innerHTML = "Feedback rating of "+trainer+" is "+feedbackrate+"/5. Given by participant-"+participant;
                   return false;
               }

        </script>
